=========================================FRAMEWORK EXPLAINATION=========================================

In my project we implemented a modular Playwright framework using a layered architecture
consisting of Tests, Tasks, Questions, Page Objects, and Utilities.

Tests describe business scenarios, tasks implement reusable user actions, questions handle assertions,
and page objects centralize UI locators.

This separation improves maintainability, reusability, and scalability.

Additionally, our framework supports API + UI validation, where backend responses are validated
along with UI behavior to ensure data consistency.

function getSmartUXUrl() {
    const originalUrl = fs.readFileSync(env.urlFile, 'utf8').trim();
    const match = originalUrl.match(/localhost:(\d+)/);

    if (!match) {
        throw new Error('Port number not found in URL');
    }

    const port = match[1];
    return `http://localhost:${port}/#!/openApp?inputPath=${env.smartuxInputPath}`;
}

export async function verifyTheComponentList(page) {
    await commonTasks.clickOnElementBygetByTitle(page, LEFT_PANE_OBJECTS.refreshComponentBtn);
    await commonTasks.clickOnElementBygetByRole(page, 'button', 'Yes');
    const url = await commonTasks.getApiUrl();
    await console.log(url)
    const apiContext = await request.newContext();
    const TARGET_SCREEN_NAMES = loadTargetScreenNames();
    const response = await apiContext.post(url, {
        data: REQUEST_PAYLOAD_GETTREE,
        headers: {"Content-Type": "application/json"},
    });
    expect(response.ok()).toBeTruthy();
    const responseBody = await response.json();
    console.log("\n Response (raw):", JSON.stringify(responseBody, null, 2));
    // ── Extract all names from the response tree ───────────────────────────
    const allNames = extractComponentNames(responseBody);
    console.log("\n All names found in response:", allNames);

    // ── Filter to only the target screen names ────────────────────────────
    const foundScreens = allNames.filter((name) =>
        TARGET_SCREEN_NAMES.includes(name)
    );
    const missingScreens = TARGET_SCREEN_NAMES.filter(
        (name) => !allNames.includes(name)
    );

    console.log("\n Target screens FOUND     :", foundScreens);
    if (missingScreens.length > 0) {
        console.warn("Target screens MISSING   :", missingScreens);
    }

    // ── Store matched screens ─────────────────────────────────────────────
    const matchedScreens = TARGET_SCREEN_NAMES.map((name) => ({
        name,
        foundInResponse: allNames.includes(name),
    }));

    console.log("\n Matched Screens Summary:");
    console.table(matchedScreens);

    // ── Assertions ────────────────────────────────────────────────────────
    expect(
        responseBody,
        "Response should not contain a jsonrpc error"
    ).not.toHaveProperty("error");

    for (const screenName of TARGET_SCREEN_NAMES) {
        expect(
            allNames,
            `Screen "${screenName}" should be present in the response tree`
        ).toContain(screenName);
    }
}


export function extractComponentNames(node, names = []) {
    if (Array.isArray(node)) {
        for (const item of node) {
            extractComponentNames(item, names);
        }
    } else if (node && typeof node === "object") {
        if (typeof node["name"] === "string") {
            names.push(node["name"]);
        }
        for (const value of Object.values(node)) {
            extractComponentNames(value, names);
        }
    }
    return names;
}
